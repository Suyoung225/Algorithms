import java.util.*;

public class Main {

    public int solution1(int[] arr1) {
        Arrays.sort(arr1);
        int smallNum = -1;
        int sum = 0;
        for (int i = 0; i < arr1.length; i++) {
            if(smallNum == -1 && arr1[i]!=i) smallNum = i;
            sum += arr1[i];
        }

        if(smallNum == -1) smallNum = 8;
        int bigNum = 45 - sum - smallNum; // 0+1+...+9 = 45
        int answer = 0;

        for (int i = smallNum; i < bigNum+1; i++) {
            answer += i;
        }

        return answer;
    }

    public String solution2(String s) {
        String[] str = s.split("");
        String answer = "";
        int count = 0;
        for(String letter : str){
            if(letter.equals(" ")){
                answer += " ";
                continue;
            }
            answer += (count%2==0) ? letter.toUpperCase() : letter.toLowerCase();
            count ++;
        }
        return answer;
    }


    public String[] solution3(String[] arr, int n) {
        HashMap<String, Integer> hm = new HashMap<>();
        for (String s : arr) {
            hm.put(s, hm.getOrDefault(s,0)+1);
        }
        ArrayList<String> arrList = new ArrayList<String>();
        for(String key : hm.keySet()){
            if(hm.get(key) == 1){
                arrList.add(key);
            }
        }
        String arr2[] = arrList.toArray(new String[arrList.size()]);

        for(int i=0; i< arr2.length;i++){
            arr2[i] =  arr2[i].charAt(n)+ arr2[i];
        }

        Arrays.sort(arr2);

        for(int i=0; i< arr2.length;i++){
            arr2[i] =  arr2[i].substring(1);
        }

        return  arr2;

    }


    public static void main(String[] args) {
        Main sol = new Main();

        // 1번
        int[] arr1 = {1,3,5,9,2,4,8,0};
        System.out.println(sol.solution1(arr1));

        //2번
        String s = "hang hae ninety nine";
        System.out.println(sol.solution2(s));


        // 3번
        String[] arr = {"coke", "water", "glass", "dog", "dog", "yogurt", "vitamin"};
        int n = 2;
        System.out.println(Arrays.toString(sol.solution3(arr, n)));


    }
}
